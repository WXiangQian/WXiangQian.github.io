class CreateSiteMapXml

    private $fileDir;

    public function __construct($filePath)

        $this->mkFolder($filePath);

        $this->fileDir = $filePath;

    /**

    public function addSiteMap($ary, $parent, $type)

        $this->writeFile($file, $nf);
    }

    public function ary2maps($idAry, $loc, $cf, $pri)
    {
        $lastmod = date('Y-m-d');
        $ary = array();

        foreach ($idAry as $id) {
            $ary[] = array(
                'loc' => $loc . $id,
                'lastmod' => $lastmod,
                'changefreq' => $cf,
                'priority' => $pri,
            );
        }

        return $ary;
    }

    //è¯»å–æ–‡ä»¶
    public function readFile($file)
    {
        if (empty($file)) {
            return false;
        }
        $f = file_exists($file);
        return $f ?: '';
    }

    //å†™å…¥æ–‡ä»¶
    public function writeFile($fileName, $content)
    {
        //echo $content;
        file_put_contents($fileName, $content); //æ›´æ–°å…¶å€¼
    }

    /**
     * é€šç”¨ maps
     * @param array $ary

     * @return string

    public function maps($ary, $parent): string

            foreach ($ary as $mval) {
                $str .= "<$parent>\r\n";
                foreach ($mval as $key => $val) {
                    $str .= "    <$key>$val</$key>\r\n";

        $rstr = substr($str, $i, strlen($str));
        $newstr = ($lstr . $substr . $rstr);
        return $newstr;
    }
